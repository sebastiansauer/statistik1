\makeatletter
\def\@part[#1]#2{%
  \ifnum \c@secnumdepth > -1
    \refstepcounter{part}%
    \addcontentsline{toc}{part}{\thepart\hspace{1em}#1}%
  \else
    \addcontentsline{toc}{part}{#1}%
  \fi
  \markboth{}{}%
  \clearpage
  \thispagestyle{empty} % <-- removes page number
  {\centering
    \interlinepenalty \@M
    \normalfont
    \ifnum \c@secnumdepth > -1
      \huge\bfseries \partname~\thepart
      \par\nobreak
      \vskip 20pt
    \fi
    \Huge \bfseries #2\par
    \vskip 40pt
  }
  \@afterheading
}
\makeatother
